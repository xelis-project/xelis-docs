    "jsonrpc": "2.0",
    "result": {

        "event_id": 42,
        "data": {
            "type": "primitive",
            "value": {
                "type": "string",

        "block_hash": "a1b2c3d4e5f60718273645566778899aabbccddeeff00112233445566778899",
